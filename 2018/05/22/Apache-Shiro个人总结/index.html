发表于 2018-05-22 |

web.xml中配置shiro的filter

在web系统中，shiro也通过filter进行拦截。filter拦截后将操作权交给spring中配置的filterChain（过虑链儿）
shiro提供很多filter。
–添加shiro过滤器

<--filter>
<--filter-name>shiroFilter
<--filter-class>org.springframework.web.filter.DelegatingFilterProxy
<--init-param>
–该值缺省为false，表示声明周期由SpringApplicationContext管理，设置为true表示ServletContainer管理
<--param-name>targetFilterLifecycle
<--param-value>true
<–/init-param>
<–/filter>
<--filter-mapping>
<--filter-name>shiroFilter
<--url-pattern>/*
<–/filter-mapping>

在Spring配置文件中：
–自定义Realm
<--bean id="myRealm" class="cn.powersystem.realm.MyRealm">
–自定义filter
<--bean id="roleOrFilter" class="cn.powersystem.realm.Cbuer">
–安全管理器
<--bean id="securityManager" class="org.apache.shiro.web.mgt.DefaultWebSecurityManager"><--property name="realm" ref="myRealm"><–/bean> 

--Shiro过滤器
<bean id="shiroFilter" class="org.apache.shiro.spring.web.ShiroFilterFactoryBean">  
    <!-- Shiro的核心安全接口,这个属性是必须的 -->  
    <property name="securityManager" ref="securityManager"/>
    <!-- 身份认证失败，则跳转到登录页面的配置 -->  
    <property name="loginUrl" value="/401.jsp"/>
    <!-- 权限认证失败，则跳转到指定页面 -->  
    <property name="unauthorizedUrl" value="/404.jsp"/>
    <!-- Shiro连接约束配置,即过滤链的定义 -->  
    <property name="filters">  
        <map>  
            <entry key="roleOrFilter" value-ref="roleOrFilter"/>  
        </map>  
    </property>  
    <property name="filterChainDefinitions">  
        <value>  
             <!-- 配置哪些页面需要受保护   以及访问这些页面需要的权限      
                 anon可以被匿名访问，或者说游客可以访问           
                 authc 必须认证之后才能访问，即登录后才能访问的页面      -->
            /login.jsp=anon
            /sys/pole/*=authc,roleOrFilter["系统管理员,线路管理员"]
            /sys/line/*=authc,roleOrFilter["系统管理员,线路管理员"]
            /sys/cord/*=authc,roleOrFilter["系统管理员,线路管理员"]
            /sys/system/*=authc,roleOrFilter["系统管理员,线路管理员"]
            /sys/task/*=authc,roleOrFilter["系统管理员,线路管理员"]
            /sys/solv/*=authc,roleOrFilter["系统管理员,线路管理员,消缺员"]
            /sys/xun/*=authc,roleOrFilter["系统管理员,线路管理员,巡检员"]
        </value>  
    </property>
</bean>

自定义Myrealm(认证与授权):
首先继承AuthorizingRealm，重写其认证与授权方法：
–授权：登录之后授权
public AuthorizationInfo doGetAuthorizationInfo(PrincipalCollection princ) {
String userCode = (String) princ.getPrimaryPrincipal();
logger.debug(“=========================”+userCode);
SimpleAuthorizationInfo authorizationInfo = new SimpleAuthorizationInfo();
authorizationInfo.setRoles(userExtensService.getRoles(userCode));
authorizationInfo.setStringPermissions(userExtensService.getPermissions(userCode));
return authorizationInfo;
}

--认证：用户验证身份
public AuthenticationInfo doGetAuthenticationInfo(
        AuthenticationToken token) throws AuthenticationException {
    String userCode  = (String) token.getPrincipal();
    logger.debug("========================="+userCode);
    Ps_User user = userExtensService.selectByUsername(userCode);
    if(user != null){
        AuthenticationInfo info = new SimpleAuthenticationInfo(
                user.getUserCode(), user.getUserPassword(), "myRealm");
        return info;
    }
    return null;
}

自定义filter：
–AuthorizationFilter抽象类事项了javax.servlet.Filter接口，它是个过滤器。
public class Cbuer extends AuthorizationFilter {

    protected boolean isAccessAllowed(ServletRequest req,
            ServletResponse resp, Object mappedValue) throws Exception {

        Subject subject = getSubject(req, resp);  
        String[] rolesArray = (String[]) mappedValue;  

        if (rolesArray == null || rolesArray.length == 0) { //没有角色限制，有权限访问  
            return true;  
        }  
        for (int i = 0; i < rolesArray.length; i++) {  
            if (subject.hasRole(rolesArray[i])) { //若当前用户是rolesArray中的任何一个，则有权限访问  
                return true;  
            }  
        }  

        return false;
    }

}
